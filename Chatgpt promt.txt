let activeState = null;
let activeRegion = 'all';

function filterRegion(region, el) {
  document.querySelectorAll('.rtab').forEach(t=>t.classList.remove('active'));
  el.classList.add('active');
  activeRegion = region;
  // Update state chips
  renderStateChips(region);
  // Hide all panels
  hideAllStatePanels();
  activeState = null;
}

function renderStateChips(region) {
  const chips = document.getElementById('stateChips');
  const states = Object.entries(NIGERIA_STATES).filter(([k,v])=>region==='all'||v.region===region);
  chips.innerHTML = states.map(([key,s])=>`
    <div class="state-chip ${activeState===key?'active':''}" onclick="selectState('${key}',this)">
      ${s.emoji} ${s.name}
    </div>`).join('');
}

function selectState(stateKey, el) {
  document.querySelectorAll('.state-chip').forEach(c=>c.classList.remove('active'));
  el.classList.add('active');
  if(activeState === stateKey) {
    // toggle off
    activeState = null;
    hideAllStatePanels();
    return;
  }
  activeState = stateKey;
  renderStatePanel(stateKey);
}

function hideAllStatePanels() {
  document.getElementById('nigeriaStatesContainer').innerHTML = '';
}

  opts.push('</optgroup>');

  const regionOrder = ['sw','ss','se','nc','fct','ne','nw'];
  regionOrder.forEach(r=>{
    const rStates = Object.entries(NIGERIA_STATES).filter(([k,v])=>v.region===r&&k!=='lagos');

      opts.push('</optgroup>');
    });
  });
  return opts.join('');
}

function doRegister(){
  const fn = document.getElementById('regFirstName').value.trim();
  const sn = document.getElementById('regSurname').value.trim();
  const email = document.getElementById('regEmail').value.trim();
  const phone = document.getElementById('regPhone').value.trim();
  const area = document.getElementById('regArea').value;
  const address = document.getElementById('regAddress').value.trim();
  const pw = document.getElementById('regPassword').value;
  const confirm = document.getElementById('regConfirm').value;

  if(!fn||!sn||!email||!phone||!area||!pw){
    alert('Please fill in all required fields.');
    return;
  }
  if(pw !== confirm){
    alert('Passwords do not match.');
    return;
  }
  if(pw.length < 8){
    alert('Password must be at least 8 characters.');
    return;
  }
  if(!email.includes('@')){
    alert('Please enter a valid email address.');
    return;
  }

  users[email] = { firstName:fn, surname:sn, email, phone, area, address, password:pw };
  document.getElementById('regSuccess').style.display='block';
  setTimeout(()=>{
    closeModal('registerModal');
    loginUser(users[email]);
    document.getElementById('regSuccess').style.display='none';
  }, 1800);
}

  const errEl = document.getElementById('loginError');

  if(users[email] && users[email].password === pw){
    errEl.style.display='none';
    closeModal('loginModal');
    loginUser(users[email]);

  }
}

  locWrap.style.display = 'block';
}

function matchItem(name){
  const n = name.toLowerCase();
  for(const key of Object.keys(PRICE_DB)){
    if(n.includes(key)) return {key, prices: PRICE_DB[key]};
  }
  // fuzzy fallback
  if(n.includes('flour')) return {key:'flour', prices:{mile12:32000,ajah:33000,oyingbo:33500,justrite:36000,ebeano:38000,blenco:37000}};
  if(n.includes('fish')) return {key:'fish', prices:{mile12:3800,ajah:3900,oyingbo:3700,justrite:5000,ebeano:5200,blenco:4900}};
  // Unknown — estimate
  return {key:'unknown', prices:{mile12:2000,ajah:2100,oyingbo:2000,justrite:2500,ebeano:2700,blenco:2600}};
}

function renderAIResults(items, location){
  const locKey = location.replace(/\s+/g,'_');
  const stores = Object.keys(STORE_META);

  // Calculate totals
  const totals = {};
  const breakdown = {};
  stores.forEach(s=>{ totals[s]=0; breakdown[s]={}; });

  items.forEach(item=>{
    const match = matchItem(item.name);
    stores.forEach(s=>{
      const price = (match.prices[s]||2000)*item.qty;
      totals[s] += price;
      breakdown[s][item.name] = {price: match.prices[s]||2000, total:price, qty:item.qty};
    });
  });

  // Score: combo of price and distance
  const scored = stores.map(s=>{
    const distRaw = STORE_META[s].dist[locKey] || STORE_META[s].dist['Lekki_Phase_1'];
    const priceScore = 1 - (totals[s] - Math.min(...stores.map(x=>totals[x]))) / Math.max(1, Math.max(...stores.map(x=>totals[x])) - Math.min(...stores.map(x=>totals[x])));
    const distScore = 1 - (distRaw - 2) / 40;
    const finalScore = Math.round((priceScore*0.65 + distScore*0.35)*100);
    return {s, total:totals[s], dist:distRaw, score:finalScore, meta:STORE_META[s], breakdown:breakdown[s]};
  }).sort((a,b)=>b.score-a.score);

  const best = scored[0];
  const html = scored.slice(0,3).map((item,i)=>{
    const rankLabels=['🏆 Best Value','🥈 Runner Up','🥉 Third Option'];
    const rankClasses=['first','second','third'];
    const itemRows = Object.entries(item.breakdown).slice(0,4).map(([name,d])=>`
      <div class="sr-item-row">
        <span class="sr-item-n">${name}${d.qty>1?' ×'+d.qty:''}</span>
        <span class="sr-item-p ${i===0?'best':''}">₦${d.total.toLocaleString()}</span>
      </div>`).join('');
    const moreItems = Object.keys(item.breakdown).length > 4 ? `<div style="font-size:.72rem;color:var(--muted);padding:4px 8px;">+ ${Object.keys(item.breakdown).length-4} more items</div>`:''
    const savings = i>0?`<div style="font-size:.72rem;color:var(--red);font-family:var(--mono);font-weight:600;">₦${(item.total-best.total).toLocaleString()} more than top pick</div>`:'<div style="font-size:.72rem;color:var(--green);font-weight:600;">✅ Cheapest option for your full list</div>';
    return `<div class="store-recommendation ${i===0?'top-pick':''}">
      <div class="sr-header">
        <div>
          <div class="sr-rank ${rankClasses[i]}">${rankLabels[i]}</div>
          <div class="sr-store-name">${item.meta.emoji} ${item.meta.name}</div>
          <span class="sr-type">${item.meta.type} · ~${item.dist}km away</span>
        </div>
        <div style="text-align:right">
          <div class="sr-total">₦${item.total.toLocaleString()}</div>
          <div class="sr-total-lbl">total · score: ${item.score}/100</div>
        </div>
      </div>
      <div class="sr-items">${itemRows}${moreItems}</div>
      ${savings}
    </div>`;
  }).join('');

  document.getElementById('storeRecs').innerHTML = html;
  document.getElementById('aiInsight').innerHTML = `<strong>🤖 AI Insight:</strong> Based on your ${items.length}-item list near <strong>${location}</strong>, <strong>${best.meta.name}</strong> offers the best total value at <strong>₦${best.total.toLocaleString()}</strong>. You'd save <strong>₦${(Math.max(...scored.map(s=>s.total))-best.total).toLocaleString()}</strong> versus the most expensive option. Consider splitting your run: buy fresh produce at the nearest open market and packaged goods at ${scored.find(s=>s.meta.type==='Supermarket')?.meta.name||'a nearby supermarket'}.`;
}
